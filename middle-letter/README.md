Acceptance Criteria
get_middle("test") # => "es"
get_middle("testing") # => "t"
get_middle("middle") # => "dd"
get_middle("A") # => "A"
get_middle("of") # => "of"


input      |      output

"A"               "A"
"of"              "of"
"testing"         "t"
"test"            "es"
"middle"          "dd"
